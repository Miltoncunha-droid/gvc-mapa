🔗 Integração Google Sheets
📋 Como configurar (uma vez só)
1. Acesse sheets.google.com e crie uma planilha chamada GVC Mapa
2. No menu, vá em Extensões → Apps Script
3. Apague o código existente e cole o código abaixo
4. Clique em Implantar → Nova implantação → App da Web
5. Em "Quem tem acesso" selecione Qualquer pessoa → Implantar
6. Copie a URL gerada e cole no campo acima
CÓDIGO DO APPS SCRIPT 
const SHEET_NAME = 'Pontos';

function doGet(e) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet()
    .getSheetByName(SHEET_NAME) || 
    SpreadsheetApp.getActiveSpreadsheet().insertSheet(SHEET_NAME);
  const data = sheet.getDataRange().getValues();
  if (data.length <= 1) return jsonResponse([]);
  const headers = data[0];
  const rows = data.slice(1).map(r => {
    const obj = {};
    headers.forEach((h,i) => obj[h] = r[i]);
    try { obj.residuos = JSON.parse(obj.residuos || '[]'); } catch(e) { obj.residuos = []; }
    return obj;
  });
  return jsonResponse(rows);
}

function doPost(e) {
  const points = JSON.parse(e.postData.contents);
  const sheet = SpreadsheetApp.getActiveSpreadsheet()
    .getSheetByName(SHEET_NAME) ||
    SpreadsheetApp.getActiveSpreadsheet().insertSheet(SHEET_NAME);
  sheet.clearContents();
  if (!points.length) return jsonResponse({ok:true});
  const headers = ['id','tipo','nome','cat','contatoNome','tel','email','obs','residuos','lat','lng','gmaps'];
  sheet.appendRow(headers);
  points.forEach(p => {
    sheet.appendRow(headers.map(h => 
      h === 'residuos' ? JSON.stringify(p[h]||[]) : (p[h]||'')
    ));
  });
  return jsonResponse({ok:true, count: points.length});
}

function jsonResponse(data) {
  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
